```typescript
-- ============================================================================
-- AURAWRITE / AUTHOR PROFILE SYSTEM
-- ============================================================================
-- Enthält:
--   1) author_profiles
--   2) author_style_sheets
--   3) author_emotion_modes
--   4) author_prompt_patterns
--   5) updated_at Triggerfunktion + Trigger
--   6) Indizes
--
-- Annahmen:
-- - extension pgcrypto ist vorhanden oder gen_random_uuid() ist verfügbar
-- - Serienzuordnung erfolgt an anderer Stelle und wird hier NICHT modelliert
-- ============================================================================


-- ============================================================================
-- 0) UPDATED_AT HELPER
-- ============================================================================

create or replace function public.set_updated_at()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;


-- ============================================================================
-- 1) AUTHOR PROFILES
-- ============================================================================
-- Zentrale Stammdaten eines Autorenprofils
-- version: Version des gesamten Profils
-- motto: kann später gezielt in Prompts injiziert werden
-- metadata: freie Erweiterungen
-- ============================================================================

create table if not exists public.author_profiles (
  id uuid primary key default gen_random_uuid(),

  name text not null,
  slug text null,
  motto text null,
  image_url text null,
  description text null,

  version integer not null default 1,

  is_active boolean not null default true,
  is_default boolean not null default false,
  sort_order integer not null default 0,

  metadata jsonb not null default '{}'::jsonb,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

comment on table public.author_profiles is
'Stammdaten eines Autorenprofils. Keine Serienzuordnung enthalten.';

comment on column public.author_profiles.name is
'Anzeigename des Autorenprofils.';

comment on column public.author_profiles.slug is
'Optionaler technischer Slug.';

comment on column public.author_profiles.motto is
'Leitsatz oder Motto des Autors, kann in Prompts an zentralen Stellen genutzt werden.';

comment on column public.author_profiles.image_url is
'Bild/Icon des Autorenprofils.';

comment on column public.author_profiles.description is
'Freitextbeschreibung des Autorenprofils.';

comment on column public.author_profiles.version is
'Version des gesamten Autorenprofils.';


-- ============================================================================
-- 2) AUTHOR STYLE SHEETS
-- ============================================================================
-- Das zentrale Stilblatt des Autors
-- injection_text: finaler Text zur direkten Prompt-Injection
-- sample_text: Beispieltexte des Autors
-- distilled_text: destillierter Stil aus Samples
-- ============================================================================

create table if not exists public.author_style_sheets (
  id uuid primary key default gen_random_uuid(),

  author_profile_id uuid not null
    references public.author_profiles(id)
    on delete cascade,

  title text not null default 'Default Style Sheet',

  injection_text text null,
  sample_text text null,
  distilled_text text null,
  notes text null,

  version integer not null default 1,

  is_active boolean not null default true,
  last_distilled_at timestamptz null,

  metadata jsonb not null default '{}'::jsonb,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

comment on table public.author_style_sheets is
'Zentrales Style Sheet eines Autorenprofils.';

comment on column public.author_style_sheets.injection_text is
'Der finale Stiltext, der direkt in Schreibprompts injiziert werden kann.';

comment on column public.author_style_sheets.sample_text is
'Schreibproben bzw. Beispieltexte des Autors.';

comment on column public.author_style_sheets.distilled_text is
'Aus Schreibproben destillierte Stilbeschreibung.';

comment on column public.author_style_sheets.last_distilled_at is
'Zeitpunkt der letzten Destillation.';


-- ============================================================================
-- 3) AUTHOR EMOTION MODES
-- ============================================================================
-- Emotionale Modi / Voice Sheets
-- Beispiele:
--   action, spannung, drama, ruhe, humor
-- Ein Datensatz pro Modus
-- ============================================================================

create table if not exists public.author_emotion_modes (
  id uuid primary key default gen_random_uuid(),

  author_profile_id uuid not null
    references public.author_profiles(id)
    on delete cascade,

  mode_key text not null,
  name text not null,
  description text null,

  injection_text text null,
  sample_text text null,
  distilled_text text null,
  notes text null,

  version integer not null default 1,

  sort_order integer not null default 0,
  is_active boolean not null default true,
  last_distilled_at timestamptz null,

  metadata jsonb not null default '{}'::jsonb,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),

  constraint author_emotion_modes_unique
    unique (author_profile_id, mode_key)
);

comment on table public.author_emotion_modes is
'Emotionale Modi bzw. Voice Sheets eines Autorenprofils.';

comment on column public.author_emotion_modes.mode_key is
'Technischer Schlüssel des Modus, z. B. action, spannung, drama, ruhe, humor.';

comment on column public.author_emotion_modes.name is
'Anzeigename des Modus.';

comment on column public.author_emotion_modes.injection_text is
'Finaler Prompt-Teil für diesen Emotionsmodus.';

comment on column public.author_emotion_modes.sample_text is
'Beispieltexte für diesen Modus.';

comment on column public.author_emotion_modes.distilled_text is
'Aus Beispieltexten destillierte Modusbeschreibung.';


-- ============================================================================
-- 4) AUTHOR PROMPT PATTERNS
-- ============================================================================
-- Wiederverwendbare Pattern-Module
-- Beispiele:
--   idea_to_chapter
--   scene_to_chapter
--   chapter_handoff
--   chapter_opening
--   scene_writing
--   scene_summary
--   chapter_summary
--   beat_generation
--   scene_generation
-- ============================================================================

create table if not exists public.author_prompt_patterns (
  id uuid primary key default gen_random_uuid(),

  author_profile_id uuid not null
    references public.author_profiles(id)
    on delete cascade,

  pattern_type text not null,
  name text not null,
  description text null,

  injection_text text null,
  sample_text text null,
  distilled_text text null,
  notes text null,

  version integer not null default 1,

  sort_order integer not null default 0,
  is_active boolean not null default true,
  last_distilled_at timestamptz null,

  metadata jsonb not null default '{}'::jsonb,

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

comment on table public.author_prompt_patterns is
'Wiederverwendbare Autoren-Pattern für Aufgaben wie Kapitelöffnung, Handoff, Summary oder Beat-Generierung.';

comment on column public.author_prompt_patterns.pattern_type is
'Technische Kategorie des Patterns, z. B. chapter_opening oder beat_generation.';

comment on column public.author_prompt_patterns.name is
'Anzeigename des Patterns.';

comment on column public.author_prompt_patterns.injection_text is
'Finaler Prompt-Teil für diesen Pattern-Typ.';

comment on column public.author_prompt_patterns.sample_text is
'Beispieltexte für diesen Pattern-Typ.';

comment on column public.author_prompt_patterns.distilled_text is
'Aus Beispieltexten destillierte Pattern-Beschreibung.';


-- ============================================================================
-- 5) TRIGGER FÜR updated_at
-- ============================================================================

drop trigger if exists trg_author_profiles_updated_at
on public.author_profiles;

create trigger trg_author_profiles_updated_at
before update on public.author_profiles
for each row
execute function public.set_updated_at();


drop trigger if exists trg_author_style_sheets_updated_at
on public.author_style_sheets;

create trigger trg_author_style_sheets_updated_at
before update on public.author_style_sheets
for each row
execute function public.set_updated_at();


drop trigger if exists trg_author_emotion_modes_updated_at
on public.author_emotion_modes;

create trigger trg_author_emotion_modes_updated_at
before update on public.author_emotion_modes
for each row
execute function public.set_updated_at();


drop trigger if exists trg_author_prompt_patterns_updated_at
on public.author_prompt_patterns;

create trigger trg_author_prompt_patterns_updated_at
before update on public.author_prompt_patterns
for each row
execute function public.set_updated_at();


-- ============================================================================
-- 6) INDIZES
-- ============================================================================

create index if not exists idx_author_profiles_name
  on public.author_profiles(name);

create index if not exists idx_author_profiles_slug
  on public.author_profiles(slug);

create index if not exists idx_author_profiles_is_active
  on public.author_profiles(is_active);


create index if not exists idx_author_style_sheets_author_profile_id
  on public.author_style_sheets(author_profile_id);

create index if not exists idx_author_style_sheets_is_active
  on public.author_style_sheets(is_active);


create index if not exists idx_author_emotion_modes_author_profile_id
  on public.author_emotion_modes(author_profile_id);

create index if not exists idx_author_emotion_modes_mode_key
  on public.author_emotion_modes(mode_key);

create index if not exists idx_author_emotion_modes_is_active
  on public.author_emotion_modes(is_active);


create index if not exists idx_author_prompt_patterns_author_profile_id
  on public.author_prompt_patterns(author_profile_id);

create index if not exists idx_author_prompt_patterns_pattern_type
  on public.author_prompt_patterns(pattern_type);

create index if not exists idx_author_prompt_patterns_is_active
  on public.author_prompt_patterns(is_active);
```
