## `public.get_matching_codex_details_for_beat(p_codex_entry_ids uuid[], p_beat_id uuid)`

### Zweck

Ermittelt die **gültigen Detail-Einträge** für einen oder mehrere Codex-Haupteinträge im Kontext eines Beats.

### Was die Funktion fachlich macht

- lädt Beat-Kontext

- berücksichtigt Exclusions

- baut einen rekursiven Relation Tree

- lädt Story-State

- filtert `codex_conditional_entries`

- wertet Bedingungen aus

- berücksichtigt:

  - `condition_mode = all`

  - `condition_mode = any`

  - `requires_pov_character_id`

  - `valid_from`

  - `valid_to`

### Aufrufparameter

| Parameter           | Typ      | Bedeutung                      |
| ------------------- | -------- | ------------------------------ |
| `p_codex_entry_ids` | `uuid[]` | Startmenge von Codex-Einträgen |
| `p_beat_id`         | `uuid`   | Beat-Kontext                   |

### Rückgabewert

| Feld                        | Typ           |
| --------------------------- | ------------- |
| `conditional_entry_id`      | `uuid`        |
| `codex_entry_id`            | `uuid`        |
| `title`                     | `text`        |
| `codex_text`                | `text`        |
| `note`                      | `text`        |
| `sort_order`                | `integer`     |
| `condition_mode`            | `text`        |
| `requires_pov_character_id` | `uuid`        |
| `valid_from`                | `timestamptz` |
| `valid_to`                  | `timestamptz` |

### Beispiel

```typescript
select *
from public.get_matching_codex_details_for_beat(
  array[
    '11111111-1111-1111-1111-111111111111'::uuid
  ],
  'aaaaaaaa-aaaa-aaaa-aaaa-aaaaaaaaaaaa'::uuid
);
```

# SQL

```typescript
create or replace function public.get_matching_codex_details_for_beat(
  p_codex_entry_ids uuid[],
  p_beat_id uuid
)
returns table(
  conditional_entry_id uuid,
  codex_entry_id uuid,
  title text,
  codex_text text,
  note text,
  sort_order integer,
  condition_mode text,
  requires_pov_character_id uuid,
  valid_from timestamptz,
  valid_to timestamptz
)
language sql
stable
as $function$
with target as (
  select
    sr.id as series_id,
    b.id as beat_id,
    ch.pov_character_entry_id as pov_character_entry_id,
    s.starts_at as effective_at,
    lower(coalesce(nullif(b.beat_text, ''), nullif(b.draft_text, ''), '')) as beat_text_normalized
  from public.beats b
  join public.scenes s
    on s.id = b.scene_id
  join public.chapters ch
    on ch.id = s.chapter_id
  join public.novels n
    on n.id = ch.novel_id
  join public.series sr
    on sr.id = n.series_id
  where b.id = p_beat_id
),
input_ids as (
  select distinct unnest(p_codex_entry_ids) as codex_entry_id
),
seed_entries as (
  select ce.id as codex_entry_id
  from input_ids i
  join public.codex_entries ce
    on ce.id = i.codex_entry_id
  where ce.is_active = true
),
seed_excluded as (
  select distinct se.codex_entry_id
  from seed_entries se
  join public.codex_exclusions ex
    on ex.entry_id = se.codex_entry_id
  join target t
    on true
  where t.beat_text_normalized like '%' || coalesce(ex.normalized_excluded_phrase, lower(ex.excluded_phrase)) || '%'
),
seed_allowed as (
  select se.codex_entry_id
  from seed_entries se
  left join seed_excluded sx
    on sx.codex_entry_id = se.codex_entry_id
  where sx.codex_entry_id is null
),
relation_tree as (
  with recursive rel as (
    select
      sa.codex_entry_id as root_entry_id,
      sa.codex_entry_id as codex_entry_id,
      array[sa.codex_entry_id] as path
    from seed_allowed sa

    union all

    select
      rel.root_entry_id,
      r.target_entry_id as codex_entry_id,
      rel.path || r.target_entry_id
    from rel
    join public.codex_relations r
      on r.source_entry_id = rel.codex_entry_id
     and r.is_active = true
     and r.pull_into_context = true
    join public.codex_entries ce
      on ce.id = r.target_entry_id
     and ce.is_active = true
    where not (r.target_entry_id = any(rel.path))
  )
  select distinct
    rel.root_entry_id,
    rel.codex_entry_id
  from rel
),
candidate_codex_ids as (
  select distinct codex_entry_id
  from relation_tree
),
candidate_excluded_roots as (
  select distinct cci.codex_entry_id
  from candidate_codex_ids cci
  join public.codex_exclusions ex
    on ex.entry_id = cci.codex_entry_id
  join target t
    on true
  where t.beat_text_normalized like '%' || coalesce(ex.normalized_excluded_phrase, lower(ex.excluded_phrase)) || '%'
),
excluded_relation_tree as (
  with recursive exrel as (
    select
      cer.codex_entry_id
    from candidate_excluded_roots cer

    union

    select
      r.target_entry_id
    from exrel
    join public.codex_relations r
      on r.source_entry_id = exrel.codex_entry_id
     and r.is_active = true
     and r.pull_into_context = true
  )
  select distinct codex_entry_id
  from exrel
),
final_codex_ids as (
  select cci.codex_entry_id
  from candidate_codex_ids cci
  left join excluded_relation_tree ert
    on ert.codex_entry_id = cci.codex_entry_id
  where ert.codex_entry_id is null
),
current_story_state as (
  select
    gss.variable_id,
    gss.variable_key,
    gss.variable_label,
    gss.value_json
  from public.get_story_state_at_beat(p_beat_id, 'gross') gss
),
candidate_entries as (
  select
    cce.id,
    cce.codex_entry_id,
    cce.title,
    cce.codex_text,
    cce.note,
    cce.sort_order,
    cce.condition_mode,
    cce.requires_pov_character_id,
    cce.valid_from,
    cce.valid_to,
    cce.series_id
  from public.codex_conditional_entries cce
  join final_codex_ids fci
    on fci.codex_entry_id = cce.codex_entry_id
  join public.codex_entries ce
    on ce.id = cce.codex_entry_id
  join target t
    on t.series_id = cce.series_id
  where cce.is_active = true
    and ce.is_active = true
    and (
      cce.requires_pov_character_id is null
      or cce.requires_pov_character_id = t.pov_character_entry_id
    )
    and (
      t.effective_at is null
      or (
        (cce.valid_from is null or cce.valid_from <= t.effective_at)
        and (cce.valid_to is null or cce.valid_to >= t.effective_at)
      )
    )
),
conditions_joined as (
  select
    ce.id as conditional_entry_id,
    coalesce(ce.condition_mode, 'all') as condition_mode,
    cond.id as condition_id,
    cond.sort_order as condition_sort_order,
    sv.id as variable_id,
    sv.type as variable_type,
    css.value_json as current_value_json,
    cond.operator,
    cond.compare_value_json,
    coalesce(
      public.evaluate_story_variable_condition(
        sv.type,
        css.value_json,
        cond.operator,
        cond.compare_value_json
      ),
      false
    ) as condition_matches
  from candidate_entries ce
  left join public.codex_conditional_entry_variable_conditions cond
    on cond.conditional_entry_id = ce.id
  left join public.story_variables sv
    on sv.id = cond.variable_id
  left join current_story_state css
    on css.variable_id = cond.variable_id
),
condition_summary as (
  select
    cj.conditional_entry_id,
    max(cj.condition_mode) as condition_mode,
    count(cj.condition_id) as total_conditions,
    coalesce(
      bool_and(cj.condition_matches) filter (where cj.condition_id is not null),
      true
    ) as all_conditions_match,
    coalesce(
      bool_or(cj.condition_matches) filter (where cj.condition_id is not null),
      false
    ) as any_condition_match
  from conditions_joined cj
  group by cj.conditional_entry_id
),
valid_entries as (
  select
    ce.*
  from candidate_entries ce
  left join condition_summary cs
    on cs.conditional_entry_id = ce.id
  where
    coalesce(cs.total_conditions, 0) = 0
    or (
      coalesce(cs.condition_mode, 'all') = 'all'
      and cs.all_conditions_match = true
    )
    or (
      coalesce(cs.condition_mode, 'all') = 'any'
      and cs.any_condition_match = true
    )
)
select
  ve.id as conditional_entry_id,
  ve.codex_entry_id,
  ve.title,
  ve.codex_text,
  ve.note,
  ve.sort_order,
  ve.condition_mode,
  ve.requires_pov_character_id,
  ve.valid_from,
  ve.valid_to
from valid_entries ve
order by ve.codex_entry_id, ve.sort_order, ve.title;
$function$;
```
